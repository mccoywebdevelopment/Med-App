   function updateWhenToTake(this,values){
    for(var i=0;i<this.whenToTake.length;++i){
        let found = false;
        for(var ix=0;ix<values.length;++ix){
            if(values[ix].toLowerCase() == this.whenToTake[i].title.toLowerCase()){
                this.whenToTake[i].value = true;
                found = true;
            }
        }
        if(!found){
            this.whenToTake[i].value = false;
        }
    }
}
